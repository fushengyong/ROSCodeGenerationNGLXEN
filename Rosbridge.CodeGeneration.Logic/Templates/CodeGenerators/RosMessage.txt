namespace RosMessages.wpf_msgs
{
    using System;
    using System.Linq;
    using Rosbridge.Client.Common.Attributes;
    using std_msgs;     

    [RosMessageType("wpf_msgs/WindowState")]
    public sealed class WindowState
    { 
        public Point2 topleft { get; set; }
        public Point2 bottomright { get; set; }
        public Single width { get; set; }
        public Single height { get; set; }

        public WindowState()
        {
        }

        public override bool Equals(object obj)
        {
            if(!(obj is WindowState))
            {
                return false;
            }
    
            WindowState other = (WindowState) obj;

            bool fieldsAreEqual = true;

            fieldsAreEqual &= this.topleft.Equals(other.topleft);
            fieldsAreEqual &= this.bottomright.Equals(other.bottomright);
            fieldsAreEqual &= this.width.Equals(other.width);
            fieldsAreEqual &= this.height.Equals(other.height);
            
            return fieldsAreEqual;
        }

        public static bool operator ==(WindowState obj1, WindowState obj2)
        {
            if (object.ReferenceEquals(null, obj1))
            {
                return object.ReferenceEquals(null, obj2);
            }

            return obj1.Equals(obj2);
        }

        public static bool operator !=(WindowState obj1, WindowState obj2)
        {
            if (object.ReferenceEquals(null, obj1))
            {
                return !object.ReferenceEquals(null, obj2);
            }

            return !obj1.Equals(obj2);
        }
    }
}
